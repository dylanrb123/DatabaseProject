CREATE TABLE Movie
    (UNIQUE movie_id        int,
            name            varchar(50),
            length          time,
            release_date    date,
            mpaa_rating     varchar(5),
            summary         varchar(500),
            trailer_url     varchar(100),
            poster_url      varchar(100),

            PRIMARY KEY (movie_id));

CREATE TABLE Genre
            (id             int,
            name            varchar(20),
            movie_id        int,
            show_id         int,

            PRIMARY KEY (id),
            FOREIGN KEY (movie_id) REFERENCES Movie,
            FOREIGN KEY (show_id) REFERENCES Show);

CREATE TABLE movie_doer
            (id             int,
            person_id       int,
            movie_id        int,

            PRIMARY KEY (id)
            FOREIGN KEY (person_id) REFERENCES Person,
            FOREIGN KEY (movie_id) REFERENCES Movie);

CREATE TABLE Person
            (person_id      int,
            name            varchar(15),
            dob             date,
            bio             varchar(500)

            PRIMARY KEY (id));

CREATE TABLE award_movie_person
            (id             int,
            award_id        int,
            movie_id        int,
            person_id       int,

            PRIMARY KEY (id),
            FOREIGN KEY (award_id) REFERENCES Award,
            FOREIGN KEY (movie_id) REFERENCES Movie,
            FOREIGN KEY (person_id) REFERENCES Person);

CREATE TABLE Award
            (award_id       int,
            isWinner        int,
            award_name      varchar(30),
            year            int,
            category        varchar(20),

            PRIMARY KEY (award_id));

CREATE TABLE award_show_person
            (id             int,
            award_id        int,
            show_id         int,
            person_id       int,

            PRIMARY KEY (id),
            FOREIGN KEY (award_id) REFERENCES Award,
            FOREIGN KEY (show_id) REFERENCES Show,
            FOREIGN KEY (person_id) REFERENCES Person);

CREATE TABLE Show
            (show_id        int,
            name            varchar(50),
            start_year      int,
            end_year        int,
            rating          varchar(5) //PG-13, etc. --we doing a user rating, like 3.5/5?
            summary         varchar(500),

            PRIMARY KEY (show_id));

CREATE TABLE show_season
            (id             int,
            show_id         int,
            season_id       int,

            PRIMARY KEY (id),
            FOREIGN KEY (show_id) REFERENCES Show,
            FOREIGN KEY (season_id) REFERENCES Season);

CREATE TABLE Season
            (season_id              int,
            season_number           int,
            start_date              date,
            end_date                date,
            poster_url              varchar(100),

            PRIMARY KEY (season_id));

CREATE TABLE season_episode
            (id             int,
            season_id       int,
            episode_id      int,

            PRIMARY KEY (id),
            FOREIGN KEY (season_id) REFERENCES Season,
            FOREIGN KEY (episode_id) REFERENCES Episode);

CREATE TABLE Episode
            (episode_id     int,
            episode_name    varchar(50),
            episode_number  int,
            length          time,
            air_date        date,
            description     varchar(100),

            PRIMARY KEY (episode_id));

CREATE TABLE show_doer
            (id             int,
            person_id       int,
            show_id         int,

            PRIMARY KEY (id)
            FOREIGN KEY (person_id) REFERENCES Person,
            FOREIGN KEY (show_id) REFERENCES Show);

CREATE TABLE review_episode
            (id             int,
            review_id       int,
            episode_id      int,

            PRIMARY KEY (id),
            FOREIGN KEY (review_id) REFERENCES Review,
            FOREIGN KEY (episode_id) REFERENCES Episode);

CREATE TABLE User_Review
            (review_id      int,
            username        varchar(30),
            star_rating     real,
            review_date     date,
            title           varchar(50),
            body            varchar(500),

            PRIMARY KEY (review_id));

CREATE TABLE review_movie
            (id             int,
            review_id       int,
            movie_id      int,

            PRIMARY KEY (id),
            FOREIGN KEY (review_id) REFERENCES Review,
            FOREIGN KEY (movie_id) REFERENCES Movie);
